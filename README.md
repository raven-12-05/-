function getSheetByName(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
  }
  return sheet;
}

// 提交「輸入頁面」資料
function submitOrder(formData) {
  try {
    var sheet = getSheetByName("排單系統");
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["時間", "暱稱/FB姓名", "委託類型", "委託字數", "付款方式", "指定日期", "委託進度"]);
    }
    
    var timestamp = new Date();
    // 強制格式化時間為純文字，避免後續讀取出錯
    var timeStr = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), "yyyy/MM/dd HH:mm:ss");
    
    var name = formData.name;
    var type = formData.type;
    var words = formData.words;
    var payment = formData.payment;
    var requiredDate = formData.requiredDate ? formData.requiredDate : "無";
    var status = "排單中";
    
    sheet.appendRow([timeStr, name, type, words, payment, requiredDate, status]);
    
    // 計算預計完稿日期 (填寫日 + 14天)
    var estDate = new Date(timestamp.getTime() + 14 * 24 * 60 * 60 * 1000);
    var estDateStr = estDate.getFullYear() + "/" + (estDate.getMonth() + 1) + "/" + estDate.getDate();
    
    return {
      success: true,
      data: {
        name: name,
        type: type,
        words: words,
        payment: payment,
        requiredDate: requiredDate,
        estDate: estDateStr,
        status: status
      }
    };
  } catch (e) {
    return { success: false, error: e.toString() };
  }
}

// 取得「排單頁面」列表 (優化：改用 getDisplayValues 避免日期物件卡死)
function getOrderList() {
  try {
    var sheet = getSheetByName("排單系統");
    var lastRow = sheet.getLastRow();
    if (lastRow <= 1) return [];
    
    // 抓取畫面上顯示的純文字值
    var values = sheet.getRange(2, 1, lastRow - 1, 7).getDisplayValues();
    var orders = [];
    
    for (var i = 0; i < values.length; i++) {
      var row = values[i];
      
      // 解析原本填寫的時間，用來推算 +14天
      var timestamp = new Date(row[0]);
      var estDateStr = "無";
      if(!isNaN(timestamp.getTime())) {
        var estDate = new Date(timestamp.getTime() + 14 * 24 * 60 * 60 * 1000);
        estDateStr = estDate.getFullYear() + "/" + (estDate.getMonth() + 1) + "/" + estDate.getDate();
      }
      
      orders.push({
        name: row[1],
        type: row[2],
        words: row[3],
        payment: row[4],
        requiredDate: row[5] ? row[5] : "無",
        estDate: estDateStr,
        status: row[6] ? row[6] : "排單中"
      });
    }
    return orders;
  } catch (e) {
    return [];
  }
}

function submitFeedback(formData) {

    var sheet = getSheetByName("委託回饋");

    sheet.appendRow([dateStr, formData.user, formData.level, formData.message]);
    return { success: true };
  } catch (e) {
    return { success: false, error: e.toString() };
  }
}

      function switchPage(pageId) {
        var pages = document.querySelectorAll('.page');
        pages.forEach(p => p.classList.remove('active'));
        
        var target = document.getElementById('page-' + pageId);
        if(target) {
          target.classList.add('active');
        }
        
        if(pageId === 'schedule') {
          fetchSchedule();
        } else if(pageId === 'feedback-view') {
          fetchFeedback();
        }
        window.scrollTo({top: 0, behavior: 'smooth'});
      }

      function openModal(title, text) {
        document.getElementById('modal-title').innerText = title;
        document.getElementById('modal-body').innerText = text;
        document.getElementById('articleModal').classList.add('show');
      }

      function closeModal() {
        document.getElementById('articleModal').classList.remove('show');
      }

      function handleOrderSubmit(e) {
        e.preventDefault();
        var formData = {
          name: document.getElementById('order-name').value,
          type: document.getElementById('order-type').value,
          words: document.getElementById('order-words').value,
          payment: document.getElementById('order-payment').value,
          requiredDate: document.getElementById('order-requiredDate').value
        };

        google.script.run.withSuccessHandler(function(response) {
          if(response.success) {
            document.getElementById('conf-name').innerText = response.data.name;
            document.getElementById('conf-type').innerText = response.data.type;
            document.getElementById('conf-words').innerText = response.data.words;
            document.getElementById('conf-payment').innerText = response.data.payment;
            document.getElementById('conf-requiredDate').innerText = response.data.requiredDate;
            document.getElementById('conf-estDate').innerText = response.data.estDate;
            document.getElementById('conf-status').innerText = response.data.status;
            
            document.getElementById('orderForm').reset();
            switchPage('confirm');
          } else {
            alert('提交失敗：' + response.error);
          }
        }).submitOrder(formData);
      }

      function loadScheduleAndGo() {
        switchPage('schedule');
      }

      function fetchSchedule() {
        var loader = document.getElementById('schedule-loading');
        var container = document.getElementById('schedule-list');
        loader.style.display = 'block';
        container.innerHTML = '';

        google.script.run.withSuccessHandler(function(orders) {
          loader.style.display = 'none';
          if(!orders || orders.length === 0) {
            container.innerHTML = '<tr><td colspan="7" class="text-center" style="padding:15px;">目前尚無排單資料。</td></tr>';
            return;
          }
          orders.forEach(function(item) {
            var tr = document.createElement('tr');
            tr.style.borderBottom = '1px solid #f0f0f0';
            var statusClass = 'status-waiting';
            
            if(item.status === '撰寫中' || item.status === '進行中' || item.status === '修改中') { 
              statusClass = 'status-doing'; 
            } else if(item.status === '已完稿' || item.status === '已交付' || item.status === '完成') { 
              statusClass = 'status-done'; 
            }
            
            tr.innerHTML = `
              <td style="padding: 15px;">${item.name}</td>
              <td style="padding: 15px;"><span class="type-tag">${item.type}</span></td>
              <td style="padding: 15px;">${item.words}</td>
              <td style="padding: 15px;">${item.payment}</td>
              <td style="padding: 15px;">${item.requiredDate}</td>
              <td style="padding: 15px;">${item.estDate}</td>
              <td style="padding: 15px;"><span class="status-badge ${statusClass}">${item.status}</span></td>
            `;
            container.appendChild(tr);
          });
        }).getOrderList();
      }

      function handleFeedbackSubmit(e) {
        e.preventDefault();
        var formData = {
          user: document.getElementById('fb-user').value,
          level: document.getElementById('fb-level').value,
          message: document.getElementById('fb-message').value
        };

        google.script.run.withSuccessHandler(function(response) {
          if(response.success) {
            alert('回饋發布成功！');
            document.getElementById('feedbackForm').reset();
            fetchFeedback(); 
          } else {
            alert('發布失敗：' + response.error);
          }
        }).submitFeedback(formData);
      }

      function fetchFeedback() {
        var loader = document.getElementById('feedback-loading');
        var container = document.getElementById('feedback-list');
        loader.style.display = 'block';
        container.innerHTML = '';

        google.script.run.withSuccessHandler(function(feedbacks) {
          loader.style.display = 'none';
          if(!feedbacks || feedbacks.length === 0) {
            container.innerHTML = '<p class="text-center" style="color:#888;">暫時還沒有任何留言唷！</p>';
            return;
          }
          feedbacks.slice().reverse().forEach(function(item) {
            var box = document.createElement('div');
            box.className = 'feedback-box';
            box.innerHTML = `
              <div class="fb-meta">
                <span class="fb-user">👤 ${item.user}</span>
                <span class="fb-date">${item.date}</span>
              </div>
              <div class="fb-level">${item.level}</div>
              <div class="fb-text">${item.message}</div>
            `;
            container.appendChild(box);
          });
        }).getFeedbackList();
      }

      window.onload = function() {
        fetchSchedule();
        fetchFeedback();
      };
    </script>
